import java.util.Scanner;
import java.util.Arrays;
public class ADestroyingTowers{
    public static void main(String[] args){
        Scanner sc= new Scanner(System.in);
        int t=sc.nextInt();
        while(t-->0){
            int n=sc.nextInt();
            int[] arr= new int[n];
            for(int i=0;i<n;i++){
                arr[i]=sc.nextInt();
            }
            for(int i=0;i<n;i++){
                for(int j=i+1;j<n;j++){
                    if(arr[i]<arr[j]){
                        arr[j]=arr[i];
                        break;
                    }
                }
            }

            int ans=Arrays.stream(arr).sum();
            System.out.println(ans);
        }
        sc.close();
    }
}
