import java.util.Scanner;

public class Main{
    public static void main(String[] args) {
        Scanner sc= new Scanner(System.in);
        int a=sc.nextInt();
        int b=sc.nextInt();
        int count=0;
        if(a==b){
            System.out.println(1);
        }
        else{
        while(a<=b){
            if(a<=b){
                a=a*3;
                b=b*2;
                count++;
            }
           

        }

        System.out.println(count);
    }
    }
}
