import java.util.Scanner;
public class ASublimeSequence {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int T= sc.nextInt();
        while(T-->0){
            int X=sc.nextInt();
            int N=sc.nextInt();
            if(N%2==0){
                System.out.println(0);
            }
            else{
                System.out.println(X);
            }
        }
        sc.close();
    }
}
