import java.util.Scanner;

public class Main{
    public static void main(String[] args) {
        Scanner sc= new Scanner (System.in);
        int p=sc.nextInt();
        int m=sc.nextInt();
        int q=sc.nextInt();
        int b=0;
        for(int i=1;i<=q;i++){
            b=b+p*i;
        }

        if(b>m){
            System.out.println(b-m);
        }
        else{
            System.out.println(0);
        }
    }
}
